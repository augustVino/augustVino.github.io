# 百度地图实现自定义InfoWindow的样式

默认的框一片白色、还有一条边框、不能圆角。。简直不能再丑。自己两年前做过一版，但是最近使用发现百度地图底层更新，一些图片的域名发生了变化，于是又修改了一下，主要用到的样式都贴上来了，可根据自己需要修改。

<!DOCTYPE html>
<html>
<head>
    <meta http-equiv="Content-Type" content="text/html; charset=utf-8" />
    <meta name="viewport" content="initial-scale=1.0, user-scalable=no" />
    <style type="text/css">
        * {
            padding: 0;
            margin: 0;
        }
        
        .BMap_cpyCtrl {
            display: none !important;
        }
        
        .anchorBL a img {
            display: none !important;
        }
        /*地图标题*/
        
        .BMap_bubble_title {
            color: white;
            font-size: 13px;
            font-weight: bold;
            text-align: left;
            padding-left: 5px;
            padding-top: 5px;
            border-bottom: 1px solid #0bb300;
            background-color: #0bb300;
        }
        /* 消息内容 */
        
        .BMap_bubble_content {
            background-color: white;
            padding-left: 5px;
            padding-top: 5px;
            padding-bottom: 10px;
        }
        /* 内容 */
        
        .BMap_pop div:nth-child(9) {
            top: 35px !important;
            border-radius: 7px;
        }
        /* 左上角删除按键 */
        
        .BMap_pop img {
            top: 43px !important;
            left: 215px !important;
        }
        
        .BMap_top {
            display: none;
        }
        
        .BMap_bottom {
            display: none;
        }
        
        .BMap_center {
            display: none;
        }
        /* 隐藏边角 */
        
        .BMap_pop div:nth-child(1) div {
            display: none;
        }
        
        .BMap_pop div:nth-child(3) {
            display: none;
        }
        
        .BMap_pop div.BMap_bubble_max_content {
            display: block !important;
            background: #fff;
            height: 18px;
            position: relative;
        }
        
        .BMap_pop div:nth-child(7) {
            display: none;
        }
        
        .BMap_pop div:nth-child(8) {
            background: url(./img/jiantou.png) no-repeat center;
            background-size: 100% 100%;
        }
        
        img[src="http://api.map.baidu.com/images/iw_close.gif"] {
            content: url('./img/iw_close.png');
        }
        
        body,
        html {
            width: 100%;
            height: 100%;
            margin: 0;
            font-family: "微软雅黑";
        }
        
        #allmap {
            width: 100%;
            height: 500px;
        }
        
        p {
            margin-left: 5px;
            font-size: 14px;
        }
    </style>
    <script type="text/javascript" src="http://api.map.baidu.com/api?v=1.2"></script>
    <title>修改百度地图信息窗口样式</title>
</head>

<body>
    <div id="allmap"></div>
    <script type="text/javascript">
        // 百度地图API功能
        var sContent =
            "<h4>天安门</h4>" +
            "<p>天安门坐落在中国北京市中心,故宫的南侧</p>" +
            "</div>";
        var opts = {
            width: 210, // 信息窗口宽度
            height: 125, // 信息窗口高度
            title: '<h4>信息窗口标题</h4>', // 信息窗口标题
            enableMessage: true, //设置允许信息窗发送短息
            message: ""
        };
        var map = new BMap.Map("allmap");
        var point = new BMap.Point(116.404, 39.915);
        var marker = new BMap.Marker(point);
        var infoWindow = new BMap.InfoWindow(sContent, opts); // 创建信息窗口对象
        map.centerAndZoom(point, 15);
        map.addOverlay(marker);
        marker.openInfoWindow(infoWindow);
    </script>
</body>
</html>
