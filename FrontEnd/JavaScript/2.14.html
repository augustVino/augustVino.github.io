// 特殊情况(#号后加了？参数)
var url = 'http://e.dangdang.com/media/h5/order/vipOrder.html#/10275/vip/vipOriginalPrice/30/shouldPayMoney/25/vipChannel/31/cId/18980/vipType/vip/autoRenew/false/strategyGrantStatus/1?channelId=70000';

function getHashParameters() {
    var arr = (location.hash || '').split('?')[1].split('&'); // location.hash = "#/10275/vip/vipOriginalPrice/30/shouldPayMoney/25/vipChannel/31/cId/18980/vipType/vip/autoRenew/false/strategyGrantStatus/1?channelId=70000"
    var params = {}
    for (var i = 0; i < arr.length; i++) {
        var data = arr[i].split('=')
        if (data.length === 2) {
            params[data[0]] = data[1]
        }
    }
    return params
}
function getHashParameter(key) {
    var params = getHashParameters()
    return params[key]
};
var cId = getHashParameter('channelId');
console.log(cId); // 70000

# Boolean Skill

# 判断数据类型

可判断类型：undefined、null、string、number、boolean、array、object、symbol、date、regexp、function、asyncfunction、arguments、set、map、weakset、weakmap、htmlparagraphelement

function DataType(tgt, type) {
    const dataType = Object.prototype.toString.call(tgt).replace(/\[object /g, "").replace(/\]/g, "").toLowerCase();
    return type ? dataType === type : dataType;
}
DataType("young"); // "string"
DataType(20190214); // "number"
DataType(true); // "boolean"
DataType([], "array"); // true
DataType({}, "array"); // false
